CREATE TABLE `clientes`.`entrevistacliente` ( `id` INT NOT NULL AUTO_INCREMENT , `user_id` INT NOT NULL , `data/hora` VARCHAR(60) NOT NULL , `educ_integral` VARCHAR(10) NOT NULL , `atend_especi` VARCHAR(10) NOT NULL , `escrito_programa` VARCHAR(10) NOT NULL , `nome_responsavel` VARCHAR(60) NOT NULL , `hab_cultura&arte` TEXT NOT NULL , `outroArte` TEXT NOT NULL,  `hab_esporte&lazer` TEXT NOT NULL , `outroEsporte` TEXT NOT NULL, `disponibilidade` VARCHAR(40) NOT NULL , `unidade_escolar` VARCHAR(40) NOT NULL , `outras_habilidades` TEXT NOT NULL , `notaExperiencia` INT(255) NOT NULL , `notaFormacao` INT(255) NOT NULL, `notaEntrevista` INT(255) NOT NULL, PRIMARY KEY (`id`),INDEX (`user_id`)) ENGINE = InnoDB;


CREATE TABLE `clientes`.`dadoscliente` ( `id` INT NOT NULL AUTO_INCREMENT , `nome` VARCHAR(60) NOT NULL , `nascimento` VARCHAR(40) NOT NULL , `endereco` TEXT NOT NULL , `fone` VARCHAR(40) NOT NULL , `rg` VARCHAR(30) NOT NULL , `cpf` VARCHAR(30) NOT NULL , `email` VARCHAR(60) NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;
